ExtendedOr
  : OrExpression
  | ExtendedOr OrExpression
  ;

OrExpression
  : AndExpression
  | OrExpression '||' AndExpression
  ;

AndExpression
  : NoExpression
  | AndExpression '&&' NoExpression
  ;

NoExpression
  : SimpleExpression
  | '!' SimpleExpression
  ;

SimpleExpression
  : WORD
  | '(' OrExpression ')'
  ;

terminals
WORD
  : /\w+/
  ;
